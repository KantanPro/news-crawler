```bash
# 配布用ディレクトリ作成
mkdir -p dist/news-crawler

# ファイルコピー（開発ファイル除外）
rsync -av --exclude-from='.distignore' . dist/news-crawler/

# ZIP パッケージ作成
cd dist
zip -r news-crawler-v2.0.0.zip news-crawler/
```
